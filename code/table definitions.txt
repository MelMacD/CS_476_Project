create table users (
username varchar(15) not null,
email varchar(30) not null,
pwd varchar(15) not null,
blogOwner bit not null,
blogName varchar(20),
primary key (email),
unique (email),
unique (username),
unique (blogName)
);

create table blog (
username varchar(15) not null,
blogName varchar(20) not null,
imageSource varchar(30),
descr varchar(300),
backgroundColor varchar(15),
font varchar(15),
primary key (blogName),
unique (username),
unique (blogName)
);

create table posts (
blogName varchar(20) not null,
id varchar(10),
topVal int,
leftVal int,
width int,
height int,
depth int,
title varchar(50),
body varchar(1000),
backgroundColor varchar(15),
fontColor varchar(15),
foreign key (blogName) references blog(blogName)
);

create table images (
blogName varchar(20) not null,
id varchar(10),
topVal int,
leftVal int,
width int,
height int,
depth int,
imageSource varchar(40),
foreign key (blogName) references blog(blogName)
);

create table videos (
blogName varchar(20) not null,
id varchar(10),
topVal int,
leftVal int,
width int,
height int,
depth int,
videoSource varchar(40),
foreign key (blogName) references blog(blogName)
);
